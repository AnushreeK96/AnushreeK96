import { AfterViewInit, Component, Input, ViewChild } from '@angular/core';
import { MatPaginator } from '@angular/material/paginator';
import { MatSort } from '@angular/material/sort';
import { MatTableDataSource } from '@angular/material/table';
import { CountryData } from 'src/app/classes/countryData';

/**
 * This component is responsible for displaying Countrywise data
 *
 * @export
 * @class CountryStatisticsComponent
 * @implements {AfterViewInit}
 */
@Component({
  selector: 'app-country-statistics',
  templateUrl: './country-statistics.component.html',
  styleUrls: ['./country-statistics.component.css']
})
export class CountryStatisticsComponent implements AfterViewInit {

  countryData: CountryData[] = [];
  dataSource: MatTableDataSource<CountryData> = new MatTableDataSource<CountryData>([]); // data for the table
  filterString: string = ''; // data entered in search bar
  displayedColumns: string[] = []; // columns displayed on table at a given time
  allColumns: string[] = []; // all the columns

  @ViewChild('paginator') paginator: MatPaginator;
  @ViewChild(MatSort) sort: MatSort;

  @Input('countryData') set inputCountryData(val) {
    if (val) {
      this.countryData = [...val];
      this.dataSource.paginator = this.paginator;
      this.dataSource = new MatTableDataSource<CountryData>(this.countryData);
      this.dataSource.paginator = this.paginator;
    }
  }


  constructor() {
     this.countryData = [];
     this.dataSource = new MatTableDataSource<CountryData>([]);
     this.filterString = '';
     this.allColumns = ['countryName', 'newConfirmed', 'totalConfirmed', 'newDeaths', 'totalDeaths', 'newRecovered', 'totalRecovered'];
     this.displayedColumns = [...this.allColumns]
  }

  /**
   * Invoked after the table view is initialized to assign the paginator and sort to the table
   *
   * @memberof CountryStatisticsComponent
   */
  ngAfterViewInit() {
    this.dataSource.paginator = this.paginator;
    this.dataSource.sort = this.sort;
  }

  /**
   * Invoked to filter the table based on searched string
   *
   * @memberof CountryStatisticsComponent
   */
  filterData() {
    this.dataSource.filter = this.filterString;
  }

  /**
   * Invoked on clicking sort on any column
   *
   * @param {*} event
   * @memberof CountryStatisticsComponent
   */
  sortData(event) {
    let colName = event.active;
    let ascending = false;
    if (event.direction === 'asc') ascending = true;
    this.countryData.sort((a, b) => this.compare(a[colName], b[colName], ascending));
    this.dataSource = new MatTableDataSource<CountryData>(this.countryData);
  }

  /**
   * Compare function used for table sorting
   *
   * @param {(number | string)} a
   * @param {(number | string)} b
   * @param {boolean} isAsc
   * @returns
   * @memberof CountryStatisticsComponent
   */
  compare(a: number | string, b: number | string, isAsc: boolean) {
    return (a < b ? -1 : 1) * (isAsc ? 1 : -1);
  }

  showColumns(...columns) {
    if (columns && columns.length) {
      this.displayedColumns = ['countryName', ...columns];
    } else {
      this.displayedColumns = [...this.allColumns]
    }
  }


}
