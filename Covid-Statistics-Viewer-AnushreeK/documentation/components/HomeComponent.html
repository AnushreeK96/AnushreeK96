import { Component, OnInit } from '@angular/core';
import { GlobalData } from 'src/app/classes/globalData';
import { CountryData } from 'src/app/classes/countryData';
import { FacadeService } from '../../services/facade.service';

/**
 * This page is the landing page of the application responsible for displaying the header,
 * tabs and invoking the child components to display the Global summary and Country data
 *
 * @export
 * @class HomeComponent
 * @implements {OnInit}
 */
@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css']
})
export class HomeComponent implements OnInit {

  globalData: GlobalData = null;
  countryData: CountryData[] = [];
  loading: boolean = true;
  currentDate: Date = null;

  constructor(private facadeService: FacadeService) {
    this.setInitialValues();
  }

  /**
   * Invoked after the constructor to fetch the data from the COVID API
   *
   * @memberof HomeComponent
   */
  ngOnInit(): void {
    this.fetchData();
  }

  /**
   * Initializes variables
   *
   * @memberof HomeComponent
   */
  setInitialValues() {
    this.globalData = null;
    this.countryData = [];
    this.loading = true;
    this.currentDate = null;
  }

  /**
   * Invokes the service to fetch the data and get the parsed Global and Country data required to be displayed by
   * passing them on to child components
   *
   * @memberof HomeComponent
   */
  fetchData() {
    this.facadeService.fetchData().subscribe((res) => {
      try {
        this.facadeService.setGlobalData(res.Global);
        this.facadeService.setCountryData(res.Countries);
        this.facadeService.setCurrentDate(res.Global);
        this.globalData = this.facadeService.getGlobalData();
        this.countryData = this.facadeService.getCountryData();
        this.currentDate = this.facadeService.getCurrentDate();
      } catch (err) {
        console.log('Error while fetching the data', err);
      }
      this.loading = false;
    }, (err) => {
      console.log('Error while fetching the data', err);
      this.loading = false;

    })
  }

  /**
   * Refreshes the page by fetching the data again
   *
   * @memberof HomeComponent
   */
  refresh() {
    this.setInitialValues();
    this.fetchData();
  }

  /**
   * Invoked to export data into an EXCEL file
   *
   * @memberof HomeComponent
   */
  exportData() {
    this.facadeService.exportAsExcelFile([{countryName: 'Global', ...this.globalData}, ...this.countryData], 'COVID-data');
  }


}
