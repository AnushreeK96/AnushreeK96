import { Injectable, Injector } from '@angular/core';
import { Observable } from 'rxjs';
import { CountryData } from '../classes/countryData';
import { GlobalData } from '../classes/globalData';
import { ApiService } from './api.service';
import { ExportService } from './export.service';

@Injectable({
  providedIn: 'root'
})
export class FacadeService {

  private _apiService: ApiService;
  private _exportService: ExportService;

  constructor(private injector: Injector) { }

  /**
   * Getter for ApiService
   *
   * @readonly
   * @type {ApiService}
   * @memberof FacadeService
   */
  private get apiService(): ApiService {
    if (!this._apiService) {
      this._apiService = this.injector.get(ApiService);
    }
    return this._apiService;
  }

  /**
   * Getter for ExportService
   *
   * @readonly
   * @private
   * @type {ExportService}
   * @memberof FacadeService
   */
  private get exportService(): ExportService {
    if (!this._exportService) {
      this._exportService = this.injector.get(ExportService);
    }
    return this._exportService;
  }

  /**
   * Invokes the HTTP GET call with COVID API
   *
   * @returns {Observable<any>}
   * @memberof FacadeService
   */
  fetchData(): Observable<any> {
    return this.apiService.fetchData();
  }

  /**
   * Parses and Sets the Global Data object
   *
   * @param {*} data
   * @returns
   * @memberof FacadeService
   */
  setGlobalData(data: any) {
    return this.apiService.setGlobalData(data);
  }

  /**
   * Returns the Global Data object
   *
   * @returns {GlobalData}
   * @memberof FacadeService
   */
  getGlobalData(): GlobalData {
    return this.apiService.getGlobalData();
  }

  /**
   * Parses and Sets the Country Data array
   *
   * @param {*} data
   * @returns
   * @memberof FacadeService
   */
  setCountryData(data: any) {
    return this.apiService.setCountryData(data);
  }

  /**
   * Returns the Country Data array
   *
   * @returns {CountryData[]}
   * @memberof FacadeService
   */
  getCountryData(): CountryData[] {
    return this.apiService.getCountryData();
  }

  /**
   * Sets the current date from the API
   *
   * @param {(string | any)} dateData
   * @returns
   * @memberof FacadeService
   */
  setCurrentDate(dateData: string | any) {
    return this.apiService.setCurrentDate(dateData);
  }

  /**
   * Returns the current Date
   *
   * @returns {Date}
   * @memberof FacadeService
   */
  getCurrentDate(): Date {
    return this.apiService.getCurrentDate();
  }

  /**
   * Exports data to EXCEL file
   *
   * @param {any[]} json
   * @param {string} excelFileName
   * @returns {void}
   * @memberof FacadeService
   */
  exportAsExcelFile(json: any[], excelFileName: string): void {
    return this.exportService.exportAsExcelFile(json, excelFileName);
  }

}
